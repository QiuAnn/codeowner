var $codeReviewPanel = $('.code-review-panel')

  $('#code_review_btn').popup ({

    position: 'bottom right',
    on: 'click'
  }).show(function(){
    list = document.querySelectorAll(".radio-item");
   //var checkedCheck = document.querySelector(".code-review-panel .checkbox-check");
     //var checkedTest = document.querySelector(".code-review-panel .checkbox-test");
    _checkedCheck = $codeReviewPanel.find(".checkbox-check");
    _checkedTest = $codeReviewPanel.find(".checkbox-test");
  })
  list.forEach(function (item) {
    item.onclick = function() {
      var checkedOption = document.querySelector(".code-review-panel .checked"); //获取选中的checked类
      checkedOption && checkedOption.classList.remove("checked"); //清除所有类
      this.classList.add("checked"); //this添加类
      if(this.classList.contains("has-checkbox")){
        $('input:checkbox').prop("checked", true);//or
        //_checkedCheck.classList.add("checked"); //清除所有类
        //_checkedTest.classList.add("checked"); 
        _checkedCheck.addClass("checked"); //清除所有类
        _checkedTest.addClass("checked"); 
        
      }else{
        _checkedCheck.removeClass("checked"); //清除所有类
        _checkedTest.removeClass("checked"); 
        $('input:checkbox').prop("checked", false);//or
      }
      
    }
  })
  _checkedCheck.on("click", function (e) {
    console.log(333333333)
    console.log(this.classList.contains("checked"))
    var checkedOption = document.querySelector(".code-review-panel .radio-checked .checked"); //获取选中的checked类
     checkedOption && checkedOption.classList.remove("checked"); //清除所有类
    if(this.classList.contains("checked")){
      this.classList.remove("checked"); 
      $("input[name='checkbox-check']").prop("checked", false);//or
    }else{
      this.classList.add("checked"); 
      $("input[name='checkbox-check']").prop("checked", true);//or
      console.log('checkedOption',checkedOption)
      if(_checkedTest.hasClass("checked")){
        var hasCheckbox = document.querySelector(".code-review-panel .radio-checked .has-checkbox"); 
        hasCheckbox.classList.add("checked") 
      }
    }
  })
  _checkedTest.on("click", function (e) {
     var checkedOption = document.querySelector(".code-review-panel .radio-checked .checked"); //获取选中的checked类
     checkedOption && checkedOption.classList.remove("checked"); //清除所有类
     if(this.classList.contains("checked")){
      this.classList.remove("checked"); 
      $("input[name='checkbox-test']").prop("checked", false);//or
    }else{
      this.classList.add("checked"); 
      $("input[name='checkbox-test']").prop("checked", true);//or
      if(_checkedCheck.hasClass("checked")){
        var hasCheckbox = document.querySelector(".code-review-panel .radio-checked .has-checkbox"); 
        hasCheckbox.classList.add("checked") 
      }
    }
  })
